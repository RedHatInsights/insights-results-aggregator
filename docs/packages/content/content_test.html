(
t
 
*
testing
.
T
)
 
{

,
 
err
 
:=

.

(
"../tests/content/ok/"
)

.

(
t
,
 
err
)

,
 
exists
 
:=

.

]

.

,

"'rule1' content is missing"
)

,

.

.

(
t
,
 
exists
,
 
"'err_key' error content is missing"
)


}
